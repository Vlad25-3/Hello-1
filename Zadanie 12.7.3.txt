
per_cent = {'ТКБ': 5.6, 'СКБ': 5.9, 'ВТБ': 4.28, 'СБЕР': 4.0} #словари - ключ:значение

prozent = list(per_cent.values())                  # перевод из словаря в списки

                                                   #print(prozent) # получем список значений 

money = int(input("Внести сумму на депозит"))/100  #вычисляем процент от внесённой суммы (целого числа)

                                                   #print(money)   # получаем 1-н процент за год 

deposit = [i * money for i in prozent]             #выполняем цикл for используя понимание списка

print(deposit)                                     #выводим накопленные средства за год вклада в каждом банке 

print(max(deposit))                                #выводим максимальное значение списка
